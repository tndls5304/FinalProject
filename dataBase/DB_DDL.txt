--ToDo 트리거
CREATE OR REPLACE TRIGGER update_todo_manager_del_yn
AFTER UPDATE OF DEL_YN ON TODO
FOR EACH ROW
BEGIN
    IF :NEW.DEL_YN = 'Y' AND :NEW.TODO_NO = 1 THEN
        UPDATE TODO_MANAGER
        SET DEL_YN = 'Y'
        WHERE TODO_NO_MAN = :NEW.TODO_NO;
    END IF;
END;
